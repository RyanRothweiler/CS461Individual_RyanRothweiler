FoodTotal = 10.
Cat, Dog = {FoodAte = 0, Happiness = 0}.

FeedAnimal = Event
	using Animal, Amount
	does FoodTotal = FoodTotal - Amount, Animal'FoodAte = Animal'FoodAte + Amount
	if FoodTotal - Amount > 0.

FeedDog = PlayerAction does FeedAnimal(Dog, 2).
FeedCat = PlayerAction does FeedAnimal(Cat, 1).
